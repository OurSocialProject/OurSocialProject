abstract class OurSocialProject {
  name: string;
  author: string;
  manifesto: string;
 
  constructor({name, author, manifesto }: {name: string;author: string;manifesto: string;}) {
    this.name = name;
    this.author = author;
    this.manifesto = manifesto;
  }
 
  abstract editMissionStatement(): string;
 
  abstract editValues(): string[];
 
  abstract editRituals(): string[];
 
  abstract shareWithWorld(): void;
}
 
class MySocialProject extends OtherSocialProject {
  constructor() {
    super({
      name: "My Social Project",
      author: "John Doe",
      manifesto:
        "MY (OUR) SOCIAL PROJECT MANIFESTO: This is what community means to me..."
    });
  }
 
  editMissionStatement() {
    return "I want to make a great community for myself...";
  }
 
  editValues() {
    return ["I want to treat others with empathy.", "..."];
  }
 
  editRituals() {
    return [
      "I will look people in the eyes.",
      "I will care about other peoples' points of view.",
      "..."
    ];
  }
 
  shareWithWorld() {
    postToFacebook(this.manifesto);
  }
}
